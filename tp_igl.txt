public class VectorHelper {
   private int v [];
   private int min ;
   private int max ;

    public void setExtrema () {
        int i=0;
        int minValue =this.v[0];
        int maxValue=this.v[0];
        while (i<v.length) {
            if (v[i] <minValue) {
                minValue=v[i];
            }
            if (v[i]> maxValue) {
                maxValue=v[i];
            }
            i++;
        }
     this.min=minValue;
        this.max=maxValue;
    }
    public void mult_scalaire (int scalaire) { //multiplie le vecteur par un scalaire
        int i=0;
//modification
        for (i=0;i<this.v.length;i++) {
         this.v[i]=v[i]*scalaire ;
        }

    }
}
