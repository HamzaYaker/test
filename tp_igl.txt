public class VectorHelper {
   private Vector<Integer> v ;
   private int min ;
   private int max ;

    public void setExtrema () {
        int i=0;
        int minValue =this.v.elementAt(0);
        int maxValue=this.v.elementAt(0);
        while (i<v.size()) {
            if (v.elementAt(i) <minValue) {
                minValue=v.elementAt(i);
            }
            if (v.elementAt(i) > maxValue) {
                maxValue=v.elementAt(i);
            }
            i++;
        }
     this.min=minValue;
        this.max=maxValue;
    }
    public void mult_scalaire (int scalaire) { //multiplie le vecteur par un scalaire
        int i=0;

        for (i=0;i<this.v.size();i++) {
         this.v.add(i ,this.v.elementAt(i)*scalaire );
        }

    }
}
